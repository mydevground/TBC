from sqlalchemy import text

SQL_SERVER = r"YOUR_SQL_SERVER_NAME"
SQL_DATABASE = "YOUR_DATABASE"
SQL_TABLE = "dbo.YourTargetTable"

# ---------------------------------------------------------
# READ CSV INTO DATAFRAME
# ---------------------------------------------------------
df = pd.read_csv(CSV_PATH, delimiter="|")   # adjust delimiter if needed
print(f"Loaded {len(df)} records from CSV.")

# ---------------------------------------------------------
# CREATE SQLALCHEMY ENGINE FOR SQL SERVER
# ---------------------------------------------------------
connection_url = sqlalchemy.engine.URL.create(
    "mssql+pyodbc",
    host=SQL_SERVER,
    database=SQL_DATABASE,
    query={"driver": "ODBC Driver 17 for SQL Server"}
)

engine = create_engine(connection_url)

# ---------------------------------------------------------
# TRUNCATE TABLE BEFORE LOAD
# ---------------------------------------------------------
with engine.begin() as conn:
    conn.execute(text(f"TRUNCATE TABLE {SQL_TABLE};"))
    print(f"Table {SQL_TABLE} truncated successfully.")

# ---------------------------------------------------------
# LOAD DATAFRAME INTO SQL SERVER
# ---------------------------------------------------------
df.to_sql(
    SQL_TABLE,
    engine,
    if_exists="append",
    index=False
)

print(f"Successfully loaded {len(df)} records into {SQL_TABLE}.")
